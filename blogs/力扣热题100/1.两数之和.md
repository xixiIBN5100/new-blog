```ts
function twoSum(nums: number[], target: number): number[] {
    const hashMap= {}
    for(let i = 0; i<nums.length; i++) {
        const complement = target - nums[i];
        if (complement in hashMap) {
            return [hashMap[complement],i]
        }
        hashMap[nums[i]] = i
    }
};
```
