```ts
function groupAnagrams(strs: string[]): string[][] {
    const map = {}

    for(const s of strs) {
        const sortedStr = s.split("").sort().join()
        if(sortedStr in map) {
            map[sortedStr].push(s)
        } else {
            map[sortedStr] = [s]
        }
    }

    return Object.values(map)
};
```
