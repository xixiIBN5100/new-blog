```ts
function longestConsecutive(nums: number[]): number {
    if(nums.length === 0) {
        return 0
    }
    const sortedNums = nums.sort((a,b) => a-b)
    let slow = 0
    let quick = 1
    let max = 1
    while(quick < nums.length) {
        if(sortedNums[quick - 1] + 1 === sortedNums[quick]) {
            quick++
        } else if (sortedNums[quick - 1] === sortedNums[quick]) {
            quick++
            slow++
        } else {
            max = Math.max(max,quick - slow)
            slow = quick
            quick++
        }
        max = Math.max(max,quick - slow)
    }
    return max
};
```
